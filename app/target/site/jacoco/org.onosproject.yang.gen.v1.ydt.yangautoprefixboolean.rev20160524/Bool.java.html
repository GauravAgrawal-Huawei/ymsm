package org.onosproject.yang.gen.v1.ydt.yangautoprefixboolean.rev20160524;

import java.util.List;
import org.onosproject.yang.gen.v1.ydt.yangautoprefixboolean.rev20160524.bool.BooleanList;

/**
 * Abstraction of an entity which represents the functionality of bool.
 */
public interface Bool {

    /**
     * Specify the node specific operation in protocols like NETCONF.
     * Applicable in protocol edit operation, not applicable in query operation
     */
    public enum OnosYangNodeOperationType {
        MERGE,
        REPLACE,
        CREATE,
        DELETE,
        REMOVE,
        NONE
    }

    /**
     * Returns the attribute onosYangNodeOperationType.
     *
     * @return value of onosYangNodeOperationType
     */
    OnosYangNodeOperationType onosYangNodeOperationType();

    /**
     * Returns the attribute booleanList.
     *
     * @return list of booleanList
     */
    List<BooleanList> booleanList();

    /**
     * Checks if the passed Bool maps the content match query condition.
     *
     * @param bool bool being passed to check for content match
     * @param isSelectAllSchemaChild is select all schema child
     * @return match result
     */
    Bool processSubtreeFiltering(Bool bool, boolean isSelectAllSchemaChild);
}

package org.onosproject.yms.app.ych.codecutils;

import org.dom4j.Element;
import org.onosproject.yms.ydt.YdtContext;

import java.util.Stack;

/**
 * Represents a single instance leaf node handler in YSB.
 */
class XmlCodecSingleInstanceLeafHandler extends XmlCodecHandler {

    @Override
    public void setXmlValue(YdtContext ydtContext, Stack<Element> domElementStack) {
        domElementStack.peek().setText(ydtContext.getValue());
        return;
    }
}
